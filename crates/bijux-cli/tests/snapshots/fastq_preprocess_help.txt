Run the FASTQ preprocess pipeline (validate → trim → filter → stats).

Usage: bijux fastq preprocess [OPTIONS]

Options:
      --list-tools
          
      --dry-run
          
      --allow-silver
          Allow silver-tier tools
      --allow-experimental
          Allow experimental-tier tools
      --pipeline-profile <PIPELINE_PROFILE>
          Pipeline profile id (default, minimal)
      --list-adapter-presets
          
      --list-adapters
          
      --env <ENV>
          
      --sample-id <SAMPLE_ID>
          
      --r1 <R1>
          
      --r2 <R2>
          
      --out <OUT>
          
      --strict
          
      --auto
          
      --objective <OBJECTIVE>
          [default: balanced] [possible values: speed, memory, retention, balanced]
      --scientific-preset <SCIENTIFIC_PRESET>
          Scientific preset profile (ancient_dna, amplicon, metagenomic, wgs_standard) [possible values: ancient_dna, amplicon, metagenomic, wgs_standard]
      --bench-corpus <BENCH_CORPUS>
          [possible values: fastq_5set]
      --allow-partial
          
      --jobs <JOBS>
          [default: 1]
      --adapter-bank-preset <ADAPTER_BANK_PRESET>
          Adapter bank preset name (default: illumina-default)
      --adapter-bank <ADAPTER_BANK>
          Adapter bank selection: preset:<name> (deprecated; use --adapter-bank-preset)
      --adapter-bank-file <ADAPTER_BANK_FILE>
          Adapter bank file (yaml/json)
      --enable-adapter <ENABLE_ADAPTER>
          
      --disable-adapter <DISABLE_ADAPTER>
          
      --polyx-preset <POLYX_PRESET>
          PolyX preset name (default: illumina_twocolor)
      --contaminant-preset <CONTAMINANT_PRESET>
          Contaminant preset name (default: illumina_default)
      --enable-contaminant-removal
          Enable contaminant k-mer removal when contaminant preset is set.
      --no-qc-post
          
      --force-merge
          
      --enable-correct
          Enable error correction stage (paired-end only)
  -h, --help
          Print help

Examples:
  bijux fastq preprocess --r1 reads.fastq.gz --out artifacts --sample-id SAMPLE
  bijux fastq preprocess --auto --objective speed --bench-corpus fastq_5set --r1 reads.fastq.gz --out artifacts --sample-id SAMPLE
  bijux fastq preprocess --list-tools
